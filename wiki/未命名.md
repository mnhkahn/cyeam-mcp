const links = [] const nodes = [] const seen = new Set() dv.pages().forEach(p => { const from = p.file.path if (!seen.has(from)) { seen.add(from) nodes.push({ id: from, name: p.file.name }) } p.file.outlinks.forEach(link => { const to = link.path if (to) { if (!seen.has(to)) { seen.add(to) nodes.push({ id: to, name: link.display }) } links.push({ source: from, target: to }) } }) }) const graph = { nodes, links } const json = JSON.stringify(graph, null, 2) const blob = new Blob([json], { type: 'application/json' }) const url = URL.createObjectURL(blob) const a = document.createElement('a') a.href = url a.download = 'obsidian-graph.json' a.click()
